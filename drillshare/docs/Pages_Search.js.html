import React from 'react';
import axios from 'axios';
import {useLocation, useNavigate} from 'react-router-dom';
import {
  Box,
  Card,
  CardActions,
  CardActionArea,
  CardContent,
  Button,
  Typography,
  Grid,
  Container,
} from '@mui/material';
import CardMedia from '@mui/material/CardMedia';

/**
 * Search
 * @return {JSX.Element}
 * @constructor
 */
const Search = () => {
  const [listings, setListing] = React.useState([]);
  const location = useLocation();
  const navigate = useNavigate();
  const data = location.state;
  // const user = location.state;

  React.useEffect(() => {
    setListing(data);
  }, [data]);

  const navigateListing = (listing) => {
    console.log(listing);
    const obj = {listing, data};
    navigate('/listing', {state: obj});
  };

  return (
    <Container>
      <Grid container>
        {listings.map((item) => (
          <Card
            variant="outlined"
            sx={{
              maxWidth: 220,
              m: 2,
              marginLeft: 'auto',
              marginRight: 'auto',
            }}
          >
            <CardActionArea onClick={() => navigateListing(item)}>
              <CardMedia
                component="img"
                maxHeight="120px"
                image={item.photos.img[0]}
                alt=""
              />

              <CardContent>
                <Typography gutterBottom variant="h5">
                  {item.title}
                </Typography>
                <Typography variant="body1" color="text.secondary">
                  {item.description}
                </Typography>
              </CardContent>

              <div></div>
            </CardActionArea>
          </Card>
        ))}
        ;
      </Grid>
    </Container>
  );
};

export default Search;
